3、购物车问题总结：
1）watch监听数组和对象的变化，代码如下

data() {
return {
shopping: [{
check: false,
img: “img/01.jpg”,
imgText: ‘太子龙男装’,
shoppingtext: ‘颜色分类 黑色’,
price: 120,
num: 0,
opr: [‘移入收藏夹’, ‘删除’]
}, {
check: false,
img: ‘./img/01.jpg’,
imgText: ‘太子龙男装’,
shoppingtext: ‘颜色分类 黑色’,
price: 199,
num: 0,
opr: [‘移入收藏夹’, ‘删除’]
}],
total_num: 0
}
}
computed: {
countTotal: function() {
let total = 0;
this.shopping.forEach(function(item) {
total += item.num
})
return total;
}
},
watch: {
shopping: {
handler: function() {
this.total_num=this.countTotal;
},
deep: true
}
}
